    <main>
      <h1>AxeOS Recovery</h1>
      <p>Please upload www.bin to recover AxeOS</p>
      <p>
        After clicking upload, please wait 60 seconds<br>
        DO NOT restart the device until response is received
      </p>

      <form id="upload-form">
        <label for="wwwfile">www.bin</label>
        <input id="wwwfile" name="wwwfile" type="file" required>
        <button id="upload" type="submit">Upload</button>
      </form>

      <button id="restart" type="button">Restart</button>

      <div class="response">
        <output id="status" aria-live="polite"></output>
        <span>Response:</span>
        <output id="response" aria-live="polite"></output>
      </div>
    </main>

    <script>
      const uploadForm = document.getElementById("upload-form");
      const fileInput = document.getElementById("wwwfile");
      const restartButton = document.getElementById("restart");
      const statusMsg = document.getElementById("status");
      const responseMsg = document.getElementById("response");

      uploadForm.addEventListener("submit", (event) => {
        event.preventDefault();
        const file = fileInput.files[0];
        if (!file) {
          statusMsg.textContent = "Error uploading!";
          responseMsg.textContent = "No www.bin selected";
          return;
        }

        statusMsg.textContent = "uploading...";
        const request = new XMLHttpRequest();
        request.open("POST", "/api/system/OTAWWW", true);
        request.setRequestHeader("Content-Type", "application/octet-stream");
        request.onload = () => {
          responseMsg.textContent = request.responseText;
          statusMsg.textContent =
            request.status === 200 ? "Upload successful!" : "Error uploading!";
        };
        request.onerror = () => {
          statusMsg.textContent = "Error uploading!";
          responseMsg.textContent = "Request failed";
        };
        request.send(file);
      });

      restartButton.addEventListener("click", () => {
        const request = new XMLHttpRequest();
        request.open("POST", "/api/system/restart", true);
        request.onload = () => {
          responseMsg.textContent = request.responseText;
        };
        request.onerror = () => {
          responseMsg.textContent = "Request failed";
        };
        request.send(null);
      });
    </script>
